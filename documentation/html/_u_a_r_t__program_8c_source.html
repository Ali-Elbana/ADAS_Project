37void MUSART_vInit( USART_InitType *A_InitStruct,
38 USART_ClockInitTypeDef *A_ClockInitStruct, USART_MemoryMapType *A_USARTx )
39{
40
41 if( A_USARTx == USART1_REG )
42 {
43
44 MRCC_vEnablePeriphralCLK( RCC_APB2, APB2ENR_USART1EN ) ;
45
46 /*configer Tx1 as alt fun*/
47 MGPIOx_ConfigType TX1 =
48 {
49 .Port = USART1_PORT , .Pin = TX1_PIN ,
50 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
51 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
52 .AF_Type = GPIOx_AF7
53 } ;
54
55 MGPIOx_vInit( &TX1 ) ;
56
57 /*configer Rx1 as alt fun*/
58 MGPIOx_ConfigType RX1 =
59 {
60 .Port = USART1_PORT , .Pin = RX1_PIN ,
61 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
62 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
63 .AF_Type = GPIOx_AF7
64 } ;
65
66 MGPIOx_vInit( &RX1 ) ;
67
68 }
69 else if( A_USARTx == USART2_REG )
70 {
71
72 MRCC_vEnablePeriphralCLK( RCC_APB2, APB1ENR_USART2EN ) ;
73
74 /*configer Tx2 as alt fun*/
75 MGPIOx_ConfigType TX2 =
76 {
77 .Port = USART2_PORT , .Pin = TX2_PIN ,
78 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
79 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
80 .AF_Type = GPIOx_AF7
81 } ;
82
83 MGPIOx_vInit( &TX2 ) ;
84
85 /*configer Rx2 as alt fun*/
86 MGPIOx_ConfigType RX2 =
87 {
88 .Port = USART2_PORT , .Pin = RX2_PIN ,
89 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
90 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
91 .AF_Type = GPIOx_AF7
92 } ;
93
94 MGPIOx_vInit( &RX2 ) ;
95
96 }
97 else if( A_USARTx == USART6_REG )
98 {
99
100 MRCC_vEnablePeriphralCLK( RCC_APB2, APB2ENR_USART6EN ) ;
101
102 /*configer Tx6 as alt fun*/
103 MGPIOx_ConfigType TX6 =
104 {
105 .Port = USART6_PORT , .Pin = TX6_PIN ,
106 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
107 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
108 .AF_Type = GPIOx_AF7
109 } ;
110
111 MGPIOx_vInit( &TX6 ) ;
112
113 /*configer Rx6 as alt fun*/
114 MGPIOx_ConfigType RX6 =
115 {
116 .Port = USART6_PORT , .Pin = RX6_PIN ,
117 .Mode = GPIOx_MODE_AF , .OutputType = GPIOx_PUSHPULL,
118 .OutputSpeed = GPIOx_MediumSpeed , .InputType = GPIOx_NoPull ,
119 .AF_Type = GPIOx_AF7
120 } ;
121
122 MGPIOx_vInit( &RX6 ) ;
123
124 }
125
126 // BuadRate/Over-sampling selections:
127 switch( A_InitStruct->Oversampling )
128 {
129
130 case OVER_SAMPLING_16:
131
132 A_USARTx->BRR_REG = UART_BRR_SAMPLING16( __PCLK__ , A_InitStruct->BaudRate ) ;
133
134 break;
135
136 case OVER_SAMPLING_8:
137
138 A_USARTx->BRR_REG = UART_BRR_SAMPLING16( __PCLK__ , A_InitStruct->BaudRate ) ;
139
140 break;
141
142 }
143
144 A_USARTx->CR1_REG = ( A_InitStruct->Oversampling << MUSART_CR1_OVER8_BIT ) |
145 ( A_InitStruct->DataWidth << MUSART_CR1_M_BIT ) |
146 ( A_InitStruct->Parity_Enable << MUSART_CR1_PCE_BIT ) |
147 ( A_InitStruct->Parity_Selection << MUSART_CR1_PS_BIT ) ;
148
149 switch (A_InitStruct->TransferDirection)
150 {
151 case TX_ONLY:
152
153 SET_BIT( A_USARTx->CR1_REG, MUSART_CR1_TE_BIT ) ;
154
155 break ;
156
157 case RX_ONLY:
158
159 SET_BIT( A_USARTx->CR1_REG, MUSART_CR1_RE_BIT ) ;
160
161 break ;
162
163 case TX_RX :
164
165 SET_BIT( A_USARTx->CR1_REG, MUSART_CR1_TE_BIT ) ;
166 SET_BIT( A_USARTx->CR1_REG, MUSART_CR1_RE_BIT ) ;
167
168 break ;
169
170 }
171
172 A_USARTx->CR2_REG = ( A_InitStruct->StopBits << MUSART_CR2_STOP_BIT ) |
173 ( A_ClockInitStruct->ClockOutput << MUSART_CR2_CLKEN_BIT ) |
174 ( A_ClockInitStruct->ClockPhase << MUSART_CR2_CPHA_BIT ) |
175 ( A_ClockInitStruct->ClockPolarity << MUSART_CR2_CPOL_BIT ) |
176 ( A_ClockInitStruct->LastBitClockPulse << MUSART_CR2_LBCL_BIT ) ;
177
178 A_USARTx->SR_REG = 0 ; // Clear all the flags.
179
180 SET_BIT( A_USARTx->CR1_REG, MUSART_CR1_UE_BIT ) ; // EN the peripheral.
181
182}

203void MUSART_vTransmitByte( USART_MemoryMapType *A_USARTx, u8_t A_u8Byte )
204{
205
206 while( GET_BIT( A_USARTx->SR_REG, MUSART_SR_TXE_BIT ) == 0 ) ;
207
208 A_USARTx->DR_REG = A_u8Byte ;
209
210 while( GET_BIT( A_USARTx->SR_REG, MUSART_SR_TC_BIT ) == 0 ) ;
211
212 // Clear the TC flag:
213 CLR_BIT( A_USARTx->SR_REG, MUSART_SR_TC_BIT ) ;
214
215}

240{
241
242 u8_t loc_u8Data = INITIAL_ZERO ;
243
244 u32_t loc_u8TimeOut = INITIAL_ZERO ;
245
246 while( (GET_BIT(A_USARTx->SR_REG, MUSART_SR_RXNE_BIT) == 0) && (loc_u8TimeOut < THRESHOLD_VALUE) )
247 {
248 loc_u8TimeOut++;
249 }
250
251 if ( loc_u8TimeOut == THRESHOLD_VALUE )
252 {
253 loc_u8Data = 255 ; // To detect the error.
