37void MUSART_vInit(USART_InitType *A_InitStruct, USART_ClockInitTypeDef *A_ClockInitStruct, USART_MemoryMapType *A_USARTx)
38{
39
40 if (A_USARTx == USART1_REG)
41 {
42
43 MRCC_vEnablePeriphralCLK(RCC_APB2, APB2ENR_USART1EN);
44
45 /*configer Tx1 as alt fun*/
46 MGPIOx_ConfigType TX1 =
47 {
48 .Port = USART1_PORT, .Pin = TX1_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
49
50 MGPIOx_vInit(&TX1);
51
52 /*configer Rx1 as alt fun*/
53 MGPIOx_ConfigType RX1 =
54 {
55 .Port = USART1_PORT, .Pin = RX1_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
56
57 MGPIOx_vInit(&RX1);
58 }
59 else if (A_USARTx == USART2_REG)
60 {
61
62 MRCC_vEnablePeriphralCLK(RCC_APB2, APB1ENR_USART2EN);
63
64 /*configer Tx2 as alt fun*/
65 MGPIOx_ConfigType TX2 =
66 {
67 .Port = USART2_PORT, .Pin = TX2_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
68
69 MGPIOx_vInit(&TX2);
70
71 /*configer Rx2 as alt fun*/
72 MGPIOx_ConfigType RX2 =
73 {
74 .Port = USART2_PORT, .Pin = RX2_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
75
76 MGPIOx_vInit(&RX2);
77 }
78 else if (A_USARTx == USART6_REG)
79 {
80
81 MRCC_vEnablePeriphralCLK(RCC_APB2, APB2ENR_USART6EN);
82
83 /*configer Tx6 as alt fun*/
84 MGPIOx_ConfigType TX6 =
85 {
86 .Port = USART6_PORT, .Pin = TX6_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
87
88 MGPIOx_vInit(&TX6);
89
90 /*configer Rx6 as alt fun*/
91 MGPIOx_ConfigType RX6 =
92 {
93 .Port = USART6_PORT, .Pin = RX6_PIN, .Mode = GPIOx_MODE_AF, .OutputType = GPIOx_PUSHPULL, .OutputSpeed = GPIOx_MediumSpeed, .InputType = GPIOx_NoPull, .AF_Type = GPIOx_AF7};
94
95 MGPIOx_vInit(&RX6);
96 }
97
98 // BuadRate/Over-sampling selections:
99 switch (A_InitStruct->Oversampling)
100 {
101
102 case OVER_SAMPLING_16:
103
104 A_USARTx->BRR_REG = UART_BRR_SAMPLING16(__PCLK__, A_InitStruct->BaudRate);
105
106 break;
107
108 case OVER_SAMPLING_8:
109
110 A_USARTx->BRR_REG = UART_BRR_SAMPLING16(__PCLK__, A_InitStruct->BaudRate);
111
112 break;
113 }
114
115 A_USARTx->CR1_REG = (A_InitStruct->Oversampling << MUSART_CR1_OVER8_BIT) |
116 (A_InitStruct->DataWidth << MUSART_CR1_M_BIT) |
117 (A_InitStruct->Parity_Enable << MUSART_CR1_PCE_BIT) |
118 (A_InitStruct->Parity_Selection << MUSART_CR1_PS_BIT);
119
120 switch (A_InitStruct->TransferDirection)
121 {
122 case TX_ONLY:
123
124 SET_BIT(A_USARTx->CR1_REG, MUSART_CR1_TE_BIT);
125
126 break;
127
128 case RX_ONLY:
129
130 SET_BIT(A_USARTx->CR1_REG, MUSART_CR1_RE_BIT);
131
132 break;
133
134 case TX_RX:
135
136 SET_BIT(A_USARTx->CR1_REG, MUSART_CR1_TE_BIT);
137 SET_BIT(A_USARTx->CR1_REG, MUSART_CR1_RE_BIT);
138
139 break;
140 }
141
142 A_USARTx->CR2_REG = (A_InitStruct->StopBits << MUSART_CR2_STOP_BIT) |
143 (A_ClockInitStruct->ClockOutput << MUSART_CR2_CLKEN_BIT) |
144 (A_ClockInitStruct->ClockPhase << MUSART_CR2_CPHA_BIT) |
145 (A_ClockInitStruct->ClockPolarity << MUSART_CR2_CPOL_BIT) |
146 (A_ClockInitStruct->LastBitClockPulse << MUSART_CR2_LBCL_BIT);
147
148 A_USARTx->SR_REG = 0; // Clear all the flags.
149
150 SET_BIT(A_USARTx->CR1_REG, MUSART_CR1_UE_BIT); // EN the peripheral.
151}

172void MUSART_vTransmitByte(USART_MemoryMapType *A_USARTx, u8_t A_u8Byte)
173{
174
175 while (GET_BIT(A_USARTx->SR_REG, MUSART_SR_TXE_BIT) == 0)
176 ;
177
178 A_USARTx->DR_REG = A_u8Byte;
179
180 while (GET_BIT(A_USARTx->SR_REG, MUSART_SR_TC_BIT) == 0)
181 ;
182
183 // Clear the TC flag:
184 CLR_BIT(A_USARTx->SR_REG, MUSART_SR_TC_BIT);
185}

207u8_t MUSART_u8ReceiveByteSynchNonBlocking(USART_MemoryMapType *A_USARTx)
208{
209
210 u8_t loc_u8Data = INITIAL_ZERO;
211
212 u32_t loc_u8TimeOut = INITIAL_ZERO;
213
214 while ((GET_BIT(A_USARTx->SR_REG, MUSART_SR_RXNE_BIT) == 0) && (loc_u8TimeOut < THRESHOLD_VALUE))
215 {
216 loc_u8TimeOut++;
217 }
218
219 if (loc_u8TimeOut == THRESHOLD_VALUE)
220 {
221 loc_u8Data = 255; // To detect the error.
222 }
223 else
224 {
225 loc_u8Data = A_USARTx->DR_REG;
226 }
227
228 return loc_u8Data;
229}

268void MUSART_vRecieveString(USART_MemoryMapType *A_USARTx, c8_t A_c8YourString[])
269{
270
271 u32_t L_u32Counter = INITIAL_ZERO;
272
273 A_c8YourString[L_u32Counter] = MUSART_u8ReceiveByteSynchBlocking(A_USARTx);
274
275 while (A_c8YourString[L_u32Counter] != '\0')
276 {
277
278 L_u32Counter++;
279
280 A_c8YourString[L_u32Counter] = MUSART_u8ReceiveByteSynchBlocking(A_USARTx);
281
282 if ((A_c8YourString[L_u32Counter] == '\n') || (A_c8YourString[L_u32Counter] == '\r'))
283 {
284 A_c8YourString[L_u32Counter] = '\0';
285 }
286 }
287}
