356u16_t MADC_u16ConvertToDigital( VAR(u8_t) A_u8ChannelNum )
357{
358
359 // Select Regular channel(1/2/3/.../16)
360 ADC->SQR3 = (ADC->SQR3 & SQ1_BIT_MANIPULATION) | ( A_u8ChannelNum ) ;
361
362 // Start conversion of regular channels:
363 SET_BIT( ADC->CR2, SWSTART ) ;
364
365 // Poll on the end of coversion flag until it's raised:
366 while( GET_BIT(ADC->SR, EOC) != FLAG_SET ) ;
367
368 // Clear the flag:
369 CLR_BIT( ADC->SR, EOC ) ;
370
371 // Return ADC Data:
372 return ADC->DR ;
373
374}
