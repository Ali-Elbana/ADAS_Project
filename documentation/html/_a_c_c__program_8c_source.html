55void AACC_vModeON( void )
56{
57
58 G_c8ACCRecievedButton = INITIAL_ZERO ;
59 Gv_u8InISR = FALSE ;
60
61 HBluetooth_vSendString( "\nACC Mode ON\n" ) ;
62
63 HBluetooth_vEnableAsynchReceive( ) ;
64
65 HBluetooth_u8AsynchReceiveByte( &Button_vCurrentData ) ;
66
67 do
68 {
69
70 HULTSNC_vTrigger( &TRIG ) ;
71
72 L_f32Distance = HULTSNC_f32GetDistance( ) ;
73
74 if( L_f32Distance < ACC_SAFE_DIST )
75 {
76 HCarMove_vStop( ) ;
77
78 HBluetooth_vSendString( SPEED0_STR ) ;
79 }
80 else if( L_f32Distance >= ACC_SAFE_DIST + HYSTRS_VALUE )
81 {
82 HCarMove_vForward( ) ;
83
84 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
85
86 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
87 }
88 else
89 {
90 /* Do Nothing */
91 }
92
93 if( Gv_u8InISR == TRUE )
94 {
95
96 Gv_u8InISR = FALSE ;
97
98 switch( G_c8ACCRecievedButton )
99 {
100
101 case EXIT_MODE_CHAR:
102
103 HCarMove_vStop( ) ;
104
105 HBluetooth_vSendString( SPEED0_STR ) ;
106
107 HBluetooth_vSendString( "\nACC Mode Off\n" ) ;
108
109 HBluetooth_u8AsynchReceiveByte( NULL ) ;
110
111 break ;
112
113 case INC_SPEED_CHAR:
114
115 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
116
117 if( L_u32SpeedValue >= SPEED_100_PERCENT )
118 {
119 L_u32SpeedValue = SPEED_100_PERCENT ;
120
121 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
122 }
123 else
124 {
125 L_u32SpeedValue += SPEED_10_PERCENT ;
126
127 HCarMove_vSpeedRatio( L_u32SpeedValue ) ;
128
129 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
130 }
131
132 break ;
133
134 case DEC_SPEED_CHAR:
135
136 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
137
138 if( L_u32SpeedValue <= SPEED_0_PERCENT )
139 {
140 L_u32SpeedValue = SPEED_0_PERCENT ;
141
142 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
143 }
144 else
145 {
146 L_u32SpeedValue -= SPEED_10_PERCENT ;
147
148 HCarMove_vSpeedRatio( L_u32SpeedValue ) ;
149
150 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
151 }
152
153 break ;
154
155 default: /* Do Nothing */ break ;
156 }
157
158 }
159 else
160 {
161 /* Do nothing */
162 }
163
164 }while( G_c8ACCRecievedButton != EXIT_MODE_CHAR ) ;
165
166}
