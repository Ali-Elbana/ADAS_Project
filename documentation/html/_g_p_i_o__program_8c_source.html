24FUNC(void) MGPIOx_vLockedPins(void)
25{
26
27 VAR(volatile u32_t) L_u32LockGPIOA = INITIAL_ZERO ;
28
29 /* Lock key write sequence */
30
31 /* WR LCKR[16] = '1' + LCKR[13,14,15] = �1� */
32 L_u32LockGPIOA = ( (1UL << LCKK_BIT_POS) | (GPIOA_PIN_POS) ) ;
33
34 GPIOA->LCKRx = L_u32LockGPIOA ;
35
36 /* WR LCKR[16] = '0' + LCKR[13,14,15] should not change*/
37 GPIOA->LCKRx = (GPIOA_PIN_POS) ;
38
39 /* WR LCKR[16] = '1'+ LCKR[13,14,15] should not change*/
40 GPIOA->LCKRx = L_u32LockGPIOA ;
41
42 /* RD LCKR */
43 L_u32LockGPIOA = GPIOA->LCKRx ;
44
46
47 VAR(volatile u32_t) L_u32LockGPIOB = INITIAL_ZERO ;

50
51 /* WR LCKR[16] = '1' + LCKR[2,3,4] = �1� */

53

56 /* WR LCKR[16] = '0' + LCKR[2,3,4] should not change*/
